// Генерация данных с притоками и оттоками (от текущей даты назад) function generateETFData(days = 30) { const data = []; let value = 0; const today = new Date(); for (let i = 0; i < days; i++) { const change = (Math.random() - 0.5) * 80; value = Math.max(-100, Math.min(100, value + change)); data.push(Math.round(value * 10) / 10); } return data; } // Обновляем подписи на реальные даты function drawETFChart(containerId, data, color, statsId) { const ctx = document.getElementById(containerId); if (!ctx) return; const context = ctx.getContext('2d'); if (etfCharts[containerId]) { etfCharts[containerId].destroy(); delete etfCharts[containerId]; } // Создаем даты от сегодня в прошлое const today = new Date(); const labels = data.map((_, i) => { const date = new Date(today); date.setDate(date.getDate() - (data.length - 1 - i)); return date.getDate() + '.' + (date.getMonth() + 1); }); const total = data.reduce((a, b) => a + b, 0); const avg = total / data.length; const max = Math.max(...data); const min = Math.min(...data); // Обновляем статистику const totalEl = document.getElementById(statsId + 'Total'); const avgEl = document.getElementById(statsId + 'Avg'); const maxEl = document.getElementById(statsId + 'Max'); const minEl = document.getElementById(statsId + 'Min'); if (totalEl) { totalEl.textContent = (total >= 0 ? '+' : '') + total.toFixed(1) + 'M'; totalEl.className = 'indicator-stat-value ' + (total >= 0 ? 'positive' : 'negative'); } if (avgEl) avgEl.textContent = '$' + avg.toFixed(1) + 'M'; if (maxEl) { maxEl.textContent = '+$' + max.toFixed(1) + 'M'; maxEl.className = 'indicator-stat-value positive'; } if (minEl) { minEl.textContent = '-$' + Math.abs(min).toFixed(1) + 'M'; minEl.className = 'indicator-stat-value negative'; } const barColors = data.map(v => v >= 0 ? color : '#f44336'); etfCharts[containerId] = new Chart(context, { type: 'bar', data: { labels: labels, datasets: [{ data: data, backgroundColor: barColors, borderColor: barColors.map(c => c), borderWidth: 1, borderRadius: 2, barPercentage: 0.8, categoryPercentage: 0.9 }] }, options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false }, tooltip: { enabled: true, callbacks: { label: function(context) { const val = context.parsed.y; return (val >= 0 ? 'Приток: +' : 'Отток: ') + val.toFixed(1) + 'M'; } } } }, scales: { x: { grid: { color: 'rgba(255,255,255,0.05)', drawBorder: false }, ticks: { color: 'rgba(255,255,255,0.2)', maxTicksLimit: 10, font: { size: 7 }, maxRotation: 45, minRotation: 45 } }, y: { grid: { color: 'rgba(255,255,255,0.08)', drawBorder: false }, ticks: { color: 'rgba(255,255,255,0.2)', callback: function(value) { return (value >= 0 ? '+' : '') + value + 'M'; }, font: { size: 8 } } } }, interaction: { intersect: false, mode: 'index' } }, plugins: [{ afterDraw: function(chart) { const yScale = chart.scales.y; const xScale = chart.scales.x; const ctx = chart.ctx; const zeroY = yScale.getPixelForValue(0); ctx.save(); ctx.beginPath(); ctx.strokeStyle = 'rgba(255,255,255,0.3)'; ctx.lineWidth = 1.5; ctx.setLineDash([4, 4]); ctx.moveTo(xScale.left, zeroY); ctx.lineTo(xScale.right, zeroY); ctx.stroke(); ctx.restore(); ctx.save(); ctx.fillStyle = 'rgba(255,255,255,0.3)'; ctx.font = '8px Arial'; ctx.textAlign = 'left'; ctx.fillText('0', xScale.right + 2, zeroY + 3); ctx.restore(); } }] }); }
