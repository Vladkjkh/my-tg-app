return null; } } async function initETFs() { console.log('Запуск initETFs()...'); // Показываем загрузку ['btcTotal', 'btcAvg', 'btcMax', 'btcMin', 'ethTotal', 'ethAvg', 'ethMax', 'ethMin'].forEach(id => { const el = document.getElementById(id); console.log('Извлеченные значения ETH:', ethValues); if (ethValues && ethValues.length > 0) { drawETFChart('ethEtf', ethValues, '#627eea', 'eth'); updateETFStats('eth', ethValues); } else { console.warn('Нет данных для ETH, использую заглушку'); updateETFStats('eth', fallbackEth); } } } catch (error) { console.error('Ошибка в initETFs:', error); } } function extractETFValues(data) { if (!data) { console.warn('Данные пустые'); return null; } console.log('Извлекаю значения из:', data); // Пробуем разные структуры ответа let flowData = null; if (data.data && data.data.flow) { flowData = data.data.flow; console.log('Нашел data.data.flow'); } else if (data.flow) { flowData = data.flow; console.log('Нашел data.flow'); } else if (Array.isArray(data)) { flowData = data; console.log('Нашел массив data'); } else if (data.data && Array.isArray(data.data)) { flowData = data.data; console.log('Нашел data.data'); } else { console.warn('Неизвестная структура данных:', Object.keys(data)); // Показываем структуру для отладки console.log('Структура:', JSON.stringify(data).substring(0, 500)); return null; } if (!flowData || flowData.length === 0) { console.warn('Нет данных flow'); return null; } console.log(`Найдено ${flowData.length} записей`); // Берем последние 30 дней const recent = flowData.slice(-30); console.log(`Последние ${recent.length} записей`); // Извлекаем числовые значения const values = recent.map(item => { if (typeof item === 'number') return item; if (item.value !== undefined) return Number(item.value); if (item.flow !== undefined) return Number(item.flow); if (item.amount !== undefined) return Number(item.amount); if (item.close !== undefined) return Number(item.close); if (item.price !== undefined) return Number(item.price); console.warn('Неизвестный формат элемента:', item); return 0; }); console.log('Извлеченные значения (первые 5):', values.slice(0, 5)); console.log('Всего значений:', values.length); // Проверяем, есть ли данные const hasData = values.some(v => v !== 0); console.log(`Есть данные: ${hasData}`); return hasData ? values : null; } function updateETFStats(coin, values) { if (!values || values.length === 0) { console.warn(`Нет значений для ${coin}`); return; } console.log(`Обновляю статистику для ${coin}:`, values); const prefix = coin.toLowerCase(); const total = values.reduce((a, b) => a + b, 0); const avg = total / values.length; const max = Math.max(...values); const min = Math.min(...values); console.log(`${coin}: total=${total}, avg=${avg}, max=${max}, min=${min}`); // Форматируем числа const formatNum = (num) => { if (Math.abs(num) >= 1e9) return (num / 1e9).toFixed(2) + 'B'; if (Math.abs(num) >= 1e6) return (num / 1e6).toFixed(2) + 'M'; if (Math.abs(num) >= 1e3) return (num / 1e3).toFixed(2) + 'K'; return num.toFixed(2); }; // Обновляем DOM const elements = { total: document.getElementById(`${prefix}Total`), avg: document.getElementById(`${prefix}Avg`), max: document.getElementById(`${prefix}Max`), min: document.getElementById(`${prefix}Min`) }; if (elements.max) elements.max.textContent = formatNum(max); if (elements.min) elements.min.textContent = formatNum(min); } // Функция для генерации тестовых данных function generateFallbackData(days) { const data = []; `; } // Запускаем при загрузке document.addEventListener('DOMContentLoaded', function() { console.log('Страница загружена, запускаю initETFs()'); initETFs(); });
