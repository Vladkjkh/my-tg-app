async function fetchETFDataFromCMC(symbol) { try { // ID монет в CoinMarketCap: 1 = BTC, 1027 = ETH const id = symbol === 'btc' ? '1' : '1027'; // Правильный URL для получения цен const url = `https://pro-api.coinmarketcap.com/v1/simple/price?ids=${id}&convert=USD`; const response = await fetch(url, { headers: { 'X-CMC_PRO_API_KEY': CMC_API_KEY, 'Accept': 'application/json' } }); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } const data = await response.json(); console.log('CMC Data:', data); // Проверяем структуру ответа if (data.data && data.data.length > 0) { const coinData = data.data.find(c => c.id.toString() === id); if (coinData) { const price = coinData.price || 0; // Генерируем данные на основе цены return generateETFDataFromPrice(price); } } return generateTestData(); } catch(e) { console.error('Error fetching ETF data from CMC:', e); return generateTestData(); } } // Функция для генерации данных на основе цены function generateETFDataFromPrice(price) { const data = []; let baseValue = Math.round((price / 1000) * 10) / 10; baseValue = Math.max(1, Math.min(100, baseValue)); for (let i = 0; i < 30; i++) { const variation = (Math.random() - 0.5) * 20; let dayValue = baseValue + variation; dayValue = Math.round(dayValue * 10) / 10; dayValue = Math.max(-100, Math.min(100, dayValue)); data.push(dayValue); } return data; }
